```typescript
import type { AnyCollection } from "./types";

// =============================================================================
// Schema Introspection
// =============================================================================

function detectSchemaLib(schema: unknown): "zod" | "valibot" | "unknown" {
  if ((schema as any)?._def?.typeName) return "zod";
  if ((schema as any)?.type && (schema as any)["~standard"]) return "valibot";
  return "unknown";
}

function zodToType(schema: unknown, depth = 0): string {
  if (depth > 10) return "unknown"; // Prevent infinite recursion

  const def = (schema as any)?._def;
  if (!def) return "unknown";

  switch (def.typeName) {
    case "ZodString": return "string";
    case "ZodNumber": return "number";
    case "ZodBoolean": return "boolean";
    case "ZodBigInt": return "bigint";
    case "ZodDate": return "Date";
    case "ZodUndefined": return "undefined";
    case "ZodNull": return "null";
    case "ZodAny": return "any";
    case "ZodUnknown": return "unknown";
    case "ZodNever": return "never";
    case "ZodLiteral": return JSON.stringify(def.value);
    case "ZodEnum": return def.values.map((v: string) => JSON.stringify(v)).join(" | ");
    case "ZodArray": return `(${zodToType(def.type, depth + 1)})[]`;
    case "ZodObject": return zodObjectToType(def.shape(), depth + 1);
    case "ZodRecord": return `Record<string, ${zodToType(def.valueType, depth + 1)}>`;
    case "ZodUnion":
    case "ZodDiscriminatedUnion":
      return def.options.map((o: unknown) => zodToType(o, depth + 1)).join(" | ");
    case "ZodIntersection":
      return `${zodToType(def.left, depth + 1)} & ${zodToType(def.right, depth + 1)}`;
    case "ZodOptional":
    case "ZodDefault":
    case "ZodNullable":
      return zodToType(def.innerType, depth + 1);
    case "ZodEffects":
      return zodToType(def.schema, depth + 1);
    default:
      return "unknown";
  }
}

function zodObjectToType(shape: Record<string, unknown>, depth: number): string {
  const fields = Object.entries(shape).map(([key, value]) => {
    const def = (value as any)?._def;
    const isOptional = def?.typeName === "ZodOptional" || def?.typeName === "ZodDefault";
    const type = zodToType(value, depth);
    const safeKey = /^[a-zA-Z_$][a-zA-Z0-9_$]*$/.test(key) ? key : JSON.stringify(key);
    return `${safeKey}${isOptional ? "?" : ""}: ${type}`;
  });

  return fields.length === 0 ? "Record<string, unknown>" : `{ ${fields.join("; ")} }`;
}

function schemaToType(schema: unknown): string {
  const lib = detectSchemaLib(schema);
  if (lib === "zod") return zodToType(schema);
  return "Record<string, unknown>";
}

// =============================================================================
// Type Generation
// =============================================================================

function toPascalCase(str: string): string {
  return str
    .split(/[-_\s]+/)
    .map((word) => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase())
    .join("");
}

export function generateTypes(collections: AnyCollection[]): string {
  const lines: string[] = [
    "// =============================================================================",
    "// Auto-generated by @usedocks/core - DO NOT EDIT",
    "// =============================================================================",
    "",
    'declare module "@usedocks/core" {',
    "  interface CollectionRegistry {",
  ];

  for (const collection of collections) {
    const dataType = schemaToType(collection.schema);
    const transformedType = collection.transform ? "Record<string, unknown>" : "undefined";

    lines.push(`    "${collection.name}": {`);
    lines.push(`      data: ${dataType};`);
    lines.push(`      transformed: ${transformedType};`);
    lines.push(`    };`);
  }

  lines.push("  }");
  lines.push("}");
  lines.push("");

  // Export entry types
  lines.push("// Entry type exports");
  for (const collection of collections) {
    const pascal = toPascalCase(collection.name);
    const dataType = schemaToType(collection.schema);
    lines.push(`export type ${pascal}Data = ${dataType};`);
  }
  lines.push("");

  return lines.join("\n");
}

export interface GeneratedOutput {
  types: string;
  collections: string[];
}

export function generate(collections: AnyCollection[]): GeneratedOutput {
  return {
    types: generateTypes(collections),
    collections: collections.map((c) => c.name),
  };
}
```

## Tests

### `packages/core/src/__tests__/typegen.test.ts`

```typescript
import { describe, it, expect } from "vitest";
import { generateTypes } from "../typegen";
import { z } from "zod";

describe("typegen", () => {
  const createCollection = (name: string, schema: any, transform?: any) => ({
    name,
    directory: `content/${name}`,
    schema,
    transform,
  });

  describe("Zod type extraction", () => {
    it("extracts string", () => {
      const col = createCollection("posts", z.object({ title: z.string() }));
      const types = generateTypes([col]);
      expect(types).toContain("title: string");
    });

    it("extracts number", () => {
      const col = createCollection("posts", z.object({ count: z.number() }));
      const types = generateTypes([col]);
      expect(types).toContain("count: number");
    });

    it("extracts boolean", () => {
      const col = createCollection("posts", z.object({ draft: z.boolean() }));
      const types = generateTypes([col]);
      expect(types).toContain("draft: boolean");
    });

    it("extracts Date", () => {
      const col = createCollection("posts", z.object({ date: z.date() }));
      const types = generateTypes([col]);
      expect(types).toContain("date: Date");
    });

    it("extracts coerced Date", () => {
      const col = createCollection("posts", z.object({ date: z.coerce.date() }));
      const types = generateTypes([col]);
      expect(types).toContain("date: Date");
    });

    it("extracts array", () => {
      const col = createCollection("posts", z.object({ tags: z.array(z.string()) }));
      const types = generateTypes([col]);
      expect(types).toContain("tags: (string)[]");
    });

    it("marks optional fields", () => {
      const col = createCollection("posts", z.object({ bio: z.string().optional() }));
      const types = generateTypes([col]);
      expect(types).toContain("bio?: string");
    });

    it("marks fields with defaults as optional", () => {
      const col = createCollection("posts", z.object({ draft: z.boolean().default(false) }));
      const types = generateTypes([col]);
      expect(types).toContain("draft?: boolean");
    });

    it("extracts enum", () => {
      const col = createCollection("posts", z.object({ status: z.enum(["draft", "published"]) }));
      const types = generateTypes([col]);
      expect(types).toContain('"draft" | "published"');
    });

    it("extracts union", () => {
      const col = createCollection("posts", z.object({ id: z.union([z.string(), z.number()]) }));
      const types = generateTypes([col]);
      expect(types).toContain("string | number");
    });

    it("extracts nullable", () => {
      const col = createCollection("posts", z.object({ avatar: z.string().nullable() }));
      const types = generateTypes([col]);
      expect(types).toContain("avatar: string");
    });

    it("extracts literal", () => {
      const col = createCollection("posts", z.object({ type: z.literal("post") }));
      const types = generateTypes([col]);
      expect(types).toContain('"post"');
    });

    it("extracts nested object", () => {
      const col = createCollection("posts", z.object({
        author: z.object({ name: z.string() }),
      }));
      const types = generateTypes([col]);
      expect(types).toContain("author: { name: string }");
    });

    it("extracts record", () => {
      const col = createCollection("posts", z.object({
        meta: z.record(z.string()),
      }));
      const types = generateTypes([col]);
      expect(types).toContain("meta: Record<string, string>");
    });
  });

  describe("module augmentation", () => {
    it("generates CollectionRegistry augmentation", () => {
      const col = createCollection("posts", z.object({ title: z.string() }));
      const types = generateTypes([col]);

      expect(types).toContain('declare module "@usedocks/core"');
      expect(types).toContain("interface CollectionRegistry");
      expect(types).toContain('"posts":');
    });

    it("includes data type in registry", () => {
      const col = createCollection("posts", z.object({ title: z.string() }));
      const types = generateTypes([col]);

      expect(types).toContain("data: { title: string }");
    });

    it("sets transformed to undefined without transform", () => {
      const col = createCollection("posts", z.object({}));
      const types = generateTypes([col]);

      expect(types).toContain("transformed: undefined");
    });

    it("sets transformed to Record with transform", () => {
      const col = createCollection("posts", z.object({}), () => ({}));
      const types = generateTypes([col]);

      expect(types).toContain("transformed: Record<string, unknown>");
    });
  });

  describe("multiple collections", () => {
    it("generates types for all collections", () => {
      const posts = createCollection("posts", z.object({ title: z.string() }));
      const pages = createCollection("pages", z.object({ slug: z.string() }));
      const types = generateTypes([posts, pages]);

      expect(types).toContain('"posts":');
      expect(types).toContain('"pages":');
    });
  });

  describe("edge cases", () => {
    it("handles empty schema", () => {
      const col = createCollection("posts", z.object({}));
      const types = generateTypes([col]);

      expect(types).toContain("data: Record<string, unknown>");
    });

    it("escapes special characters in keys", () => {
      const col = createCollection("posts", z.object({ "my-field": z.string() }));
      const types = generateTypes([col]);

      expect(types).toContain('"my-field": string');
    });

    it("generates header comment", () => {
      const col = createCollection("posts", z.object({}));
      const types = generateTypes([col]);

      expect(types).toContain("Auto-generated");
      expect(types).toContain("DO NOT EDIT");
    });
  });
});
```
